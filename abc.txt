extern function solve(a, b)
{
var d = WMLBrowser.getVar("op");
var z = 0;

if (d == "+"){
    z = a + b;
  }
else if (d == "-"){
    z = a - b;
  }
else if (d == "*"){
    z = a * b;
  }
else if (d == "/"){
    z = a / b;
  }
WMLBrowser.setVar("r", z);
WMLBrowser.go("#result");
WMLBrowser.refresh();
}
